Mandatory.

- Case 1:

< Makefile cat | grep srcs > out1
./pipex Makefile cat "grep srcs" out2

- Case 2:

< non_existent_file cat | grep pattern > out1
./pipex non_existent_file cat "grep pattern" out2

- Case 3:

< Makefile catsd | ls > out1
./pipex Makefile catsd ls out2

- Case 4:

< Makefile cat | lspf > out1
./pipex Makefile cat lspf out2

- Case 5:

< Makefile catsd | lspf > out1
./pipex Makefile catsd lspf out2

- Case 6:

< locked cat | ls > out1
./pipex locked cat ls out2

- Case 7:

< /dev/stdin cat | grep 123 > out1
./pipex /dev/stdin cat "grep 123" out2

- Case 8:

< /dev/stdin cat | grep 123 > /dev/stdout
./pipex /dev/stdin cat "grep 123" /dev/stdout

- Case 9:

< Makefile /bin/caqwdt  | /bin/lsqwd -ls > out1
./pipex Makefile /bin/caqwdt "/bin/lsqwd -ls" out2

- Case 10:

< Makefile ./a.out | ls > out1
./pipex Makefile ./a.out ls out2

- Case 11:

< Makefile /bin/cat | /bin/ls -ls > out1
./pipex Makefile /bin/cat "/bin/ls -ls" out2
